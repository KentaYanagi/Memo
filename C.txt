変数 = 数値や文字などを格納するための箱
整数を格納するための箱: int型

int n; 宣言 型がintでnの変数
int: -2147483648(2^31) ~ 2147483647

printf("b=%d c=%d\n", b, c);
%d:10進数での読み込み

scanf("%d", &n);

#include <stdio.h>

int main(void){
    int n;
    printf("Type in an integer;");
    scanf("%d", &n);        %d 10進数で読み込み &代入したい変数
    printf("n = %d\n", n); 
    return 0;
}
